    filename ft15f001 clear;

    /*----                                                                   ----*/
    /*---- records 144 bytes long hold 12 x,y,z triplets                     ----*/
    /*---- 8 records hold 96 values                                          ----*/
    /*---- filename also eliminates problems with end of record character    ----*/
    /*----                                                                   ----*/

    filename ft15f001 temp lrecl=144 recfm=f;
    data have41;
      infile ft15f001 ;
      input (x y z) (3*4.3)  @@ ;
